创建page/componet/provider

// **需要到app.module.ts中注册**
// 创建component
$ ionic g component QicklyComponent --componentsDir src/pages/home

// 创建page
$ ionic g page NewsDetailPage --pagesDir src/pages/news

// 创建service
$ ionic g provider HttpClient --providersDir src/providers

iOS、Android 打包

// 添加 iOS 支持
$ ionic platform add ios

// 编译iOS工程
$ ionic build ios

// 运行到iOS模拟器
$ ionic emulate ios

// 运行到iOS真机
$ ionic run ios

// 上传到蒲公英、AppStore
$ cd platform/ios && fastlane ios adhoc comments:"YOUR CHANGELOG"

// 添加 Android 支持
$ ionic platform add ios

// 编译Android工程
$ ionic build android

// 运行到Android模拟器
$ ionic emulate android

// 运行到Android真机
$ ionic run android

Angular 表达式

// 属性、事件 绑定
// `heroImageUrl`是变量
<img [src]="heroImageUrl">、<img src="{{heroImageUrl}}">
<div (tap)="this.activityItemClick.emit(1)"></div>

// 隐藏/显示/移除 元素，**区别**
<ion-infinite-scroll [style.display]="hasMore ? 'block' : 'none'"></ion-infinite-scroll>
<ion-infinite-scroll [class.hidden]="!hasMore"></ion-infinite-scroll>
// 移除
<ion-infinite-scroll *ngIf="hasMore"></ion-infinite-scroll>

// NgSwitch
<div [ngSwitch]="mySeg">
    <div *ngSwitchCase="'news'">...</div>
    <div *ngSwitchCase="'notice'">...</div>
    <div *ngSwitchCase="'activity'">...</div>
</div>

// NgFor
<div *ngFor="let activity of activityList; let i = index;" (tap)="this.activityItemClick.emit(activity)">...</div>
